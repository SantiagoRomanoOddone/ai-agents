1. Create the environment
    cd ai-agents/Labfiles/07-use-agent-tools-with-mcp/Python
    python -m venv labenv

2. Activate the environment
    .\labenv\Scripts\activate.bat or ./labenv/bin/Activate.ps1

3. Install requirements
    pip install -r requirements.txt --pre azure-ai-projects mcp

4. Add endpoint to .env

5. Login to AZ 
    az login

6. Run the app
    python client.py
